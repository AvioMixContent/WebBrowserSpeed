                edge: 18.3,
                chrome: 17.3,
                firefox: 14.9,
                brave: 17.1,

                safari: 0.0
            },
            'config4': { // i7 10750H, RTX 2070 Super laptop, 16 GB DDR5 2666 MHz
                edge: 19.0,

                firefox: 15.5,
                brave: 18.7,

                safari: 0.0
            }

        /**
         * Updates the displayed browser speeds based on the provided data.
         * @param {object} data - An object containing scores for each browser.
         */
        function updateBrowserSpeeds(data) {
            const maxScore = 50; // Maximum score for scaling the bars

            // Update Edge browser data
            document.querySelector('.browser-bar-vertical.edge').style.height = `calc((${data.edge} / ${maxScore}) * 100%)`;
            document.querySelector('.browser-card.border-edge span:first-of-type').textContent = data.edge.toFixed(1);

            // Update Chrome browser data
            document.querySelector('.browser-bar-vertical.chrome').style.height = `calc((${data.chrome} / ${maxScore}) * 100%)`;
            document.querySelector('.browser-card.border-chrome span:first-of-type').textContent = data.chrome.toFixed(1);

            // Update Firefox browser data
            document.querySelector('.browser-bar-vertical.firefox').style.height = `calc((${data.firefox} / ${maxScore}) * 100%)`;
            document.querySelector('.browser-card.border-firefox span:first-of-type').textContent = data.firefox.toFixed(1);

            // Update Brave browser data
            document.querySelector('.browser-bar-vertical.brave').style.height = `calc((${data.brave} / ${maxScore}) * 100%)`;
            document.querySelector('.browser-card.border-brave span:first-of-type').textContent = data.brave.toFixed(1);

            // Update Opera browser data
            document.querySelector('.browser-bar-vertical.opera').style.height = `calc((${data.opera} / ${maxScore}) * 100%)`;
            document.querySelector('.browser-card.border-opera span:first-of-type').textContent = data.opera.toFixed(1);

            // Select Safari elements robustly to avoid errors if classes change
            const safariBar = document.querySelector('.browser-card.border-safari-na .browser-bar-vertical.safari-na') || document.querySelector('.browser-card.border-safari .browser-bar-vertical.safari');
            const safariCard = document.querySelector('.browser-card.border-safari-na') || document.querySelector('.browser-card.border-safari');
            const safariScoreSpan = safariCard ? safariCard.querySelector('span:first-of-type') : null;
            const safariTextSpan = safariCard ? safariCard.querySelector('span:last-of-type') : null;

            // Update Safari browser data based on availability
            if (safariBar && safariScoreSpan && safariTextSpan && safariCard) {
                if (data.safari > 0) {
                    safariBar.classList.remove('safari-na');
                    safariBar.classList.add('safari');
                    safariBar.style.height = `calc((${data.safari} / ${maxScore}) * 100%)`;
                    safariScoreSpan.textContent = data.safari.toFixed(1);
                    safariTextSpan.textContent = 'points';
                    safariCard.classList.remove('border-safari-na');
                    safariCard.classList.add('border-safari');
                } else {
                    safariBar.classList.remove('safari');
                    safariBar.classList.add('safari-na');
                    safariBar.style.height = `calc((${data.safari} / ${maxScore}) * 100%)`;
                    safariScoreSpan.textContent = data.safari.toFixed(1);
                    safariTextSpan.textContent = '(Not available / Not tested)';
                    safariCard.classList.remove('border-safari');
                    safariCard.classList.add('border-safari-na');
                }
            } else {
                console.error("One or more Safari elements not found for update.");
            }
        }

        /**
         * Handles the selection of a PC configuration and updates the UI accordingly.
         * @param {string} configKey - The key of the selected configuration (e.g., 'config1').
         */
        function selectConfiguration(configKey) {
            console.log("Selected PC configuration:", configKey);

            // Reset all buttons to their default (non-active) state and apply correct border colors
            document.querySelectorAll('nav button').forEach(button => {
                button.classList.remove('bg-blue-500', 'bg-green-500', 'text-white'); // Remove active styles and fill colors
                button.classList.add('bg-white', 'text-gray-800'); // Apply default text/background

                // Remove any specific border colors
                button.classList.remove('border-green-500', 'border-blue-500');

                // Reapply the correct border color based on its configKey
                const currentButtonConfigKey = button.getAttribute('onclick').match(/'([^']+)'/)[1];
                if (currentButtonConfigKey === 'config2') {
                    button.classList.add('border-green-500');
                } else {
                    button.classList.add('border-blue-500');
                }
            });

            // Apply active styles to the clicked button
            const selectedButton = document.querySelector(`button[onclick="selectConfiguration('${configKey}')"]`);
            if (selectedButton) {
                selectedButton.classList.remove('bg-white', 'text-gray-800'); // Remove default text/background
                selectedButton.classList.add('text-white'); // Apply active text color

                // Apply fill color based on its configKey
                if (configKey === 'config2') {
                    selectedButton.classList.add('bg-green-500');
                } else {
                    selectedButton.classList.add('bg-blue-500');
                }
            }

            // Update the browser speeds based on the selected configuration data
            if (configurations[configKey]) {
                updateBrowserSpeeds(configurations[configKey]);
            }
        }

        // Initial setup on page load
        document.addEventListener('DOMContentLoaded', () => {
            // Apply correct borders and initial state to all buttons
            document.querySelectorAll('nav button').forEach(button => {
                button.classList.add('border-4'); // Make borders bold
                const buttonConfigKey = button.getAttribute('onclick').match(/'([^']+)'/)[1];
                if (buttonConfigKey === 'config2') {
                    button.classList.add('border-green-500');
                } else {
                    button.classList.add('border-blue-500');
                }
            });

            // Then simulate click on the first button to set its active state
            const firstButton = document.querySelector('nav button');
            if (firstButton) {
                const onclickAttr = firstButton.getAttribute('onclick');
                const configKeyMatch = onclickAttr.match(/'([^']+)'/);
                if (configKeyMatch && configKeyMatch[1]) {
                    selectConfiguration(configKeyMatch[1]);
                }
            }
        });
